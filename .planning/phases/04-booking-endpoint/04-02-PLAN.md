];
function setCors(req, res) {
  const origin = req.headers.origin || '';
  const allowed = ALLOWED_ORIGINS.find(o => origin.startsWith(o));
  res.setHeader('Access-Control-Allow-Origin', allowed || '*');
  res.setHeader('Access-Control-Allow-Methods', 'POST, OPTIONS');
  res.setHeader('Access-Control-Allow-Headers', 'Content-Type');
}

// api/book.js
// POST /api/book
// Requirements: BOOK-01, BOOK-02, BOOK-03

import {guestyFetch} from '../lib/guesty.js';
import {UPSELLS} from '../lib/upsells-config.js';
import {sendUpsellNotification} from '../lib/notify.js';

const FALLBACK_URL = 'https://svpartners.guestybookings.com/en/properties/693366e4e2c2460012d9ed96';
const INSTANT_URL = (quoteId) => `https://booking.guesty.com/api/reservations/quotes/${quoteId}/instant`;
const ALLOWED_ORIGINS = ['https://mattgshepard-prog.github.io', 'http://localhost:3000', 'http://localhost:8080'];

function setCors(req, res) — copy verbatim from api/quote.js

export default async function handler(req, res):
  1. setCors + OPTIONS guard + POST method guard (405 if not POST)
  2. Destructure body: { quoteId, ratePlanId, ccToken, checkIn, checkOut, guest, upsells }
  3. Validate required fields in this order:
     - quoteId missing → 400
     - ccToken missing → 400
     - ccToken not starting with 'pm_' → 400 (use !ccToken.startsWith('pm_'))
     - checkIn or checkOut missing → 400
     - guest missing → 400
     - guest.firstName, guest.lastName, guest.email, guest.phone any missing → 400
  4. Upsell validation:
     const upsellIds = Array.isArray(upsells) ? upsells : [];
     const validIds = new Set(UPSELLS.map(u => u.id));
     const unknown = upsellIds.filter(id => !validIds.has(id));
     if (unknown.length > 0) return 400 with error message including unknown IDs
  5. Enrich upsells and compute total:
     const enrichedUpsells = upsellIds.map(id => UPSELLS.find(u => u.id === id));
     const upsellTotal = enrichedUpsells.reduce((sum, u) => sum + u.price, 0);
  6. Build Guesty request body:
     const body = {ccToken, guest: {firstName, lastName, email, phone}};
     if (ratePlanId) body.ratePlanId = ratePlanId;
  7. try { const resp = await guestyFetch(INSTANT_URL(quoteId), {method: 'POST', body: JSON.stringify(body)}); }
  8. Error mapping (inside if (!resp.ok)):
     - resp.status === 410 → 410 QUOTE_EXPIRED
     - resp.status === 402 || resp.status === 422 → 402 PAYMENT_DECLINED
     - else → 500 BOOKING_FAILED
     All error responses include fallbackUrl.
  9. Parse success: const data = await resp.json();
     const reservationId = data._id;
     const confirmationCode = data.confirmationCode;
  10. Fire-and-forget notification (NO await):
      if (enrichedUpsells.length > 0) {
        sendUpsellNotification({guest, checkIn, checkOut, confirmationCode, upsells: enrichedUpsells});
      }
  11. Return 200: { success: true, reservationId, confirmationCode, status: 'confirmed', upsells: upsellIds, upsellTotal }
  12. Outer catch → 500 BOOKING_FAILED with fallbackUrl
